     1	import axios from 'axios';
     2	import AsyncStorage from '@react-native-async-storage/async-storage';
     3	
     4	const api = axios.create({
     5	  baseURL: 'https://rh.lhsolucao.com.br',
     6	});
     7	
     8	// Interceptor para injetar o Token em cada chamada automaticamente
     9	api.interceptors.request.use(async (config) => {
    10	  const token = await AsyncStorage.getItem('@LH_Token');
    11	  if (token) {
    12	    config.headers.Authorization = `Bearer ${token}`;
    13	  }
    14	  return config;
    15	});
    16	
    17	export default api;
