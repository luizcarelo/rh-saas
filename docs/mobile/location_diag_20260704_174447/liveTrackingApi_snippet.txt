    72	const LIVE_TRACKING_DEVICE_UID_KEY = '@LH_LiveTrackingDeviceUid';
    73	
    74	function generateDeviceUid() {
    75	  return (
    76	    'mobile-' +
    77	    Date.now().toString(36) +
    78	    '-' +
    79	    Math.random().toString(36).slice(2, 10)
    80	  );
    81	}
    82	
    83	export async function getOrCreateMobileDeviceUid() {
    84	  const existing = await AsyncStorage.getItem(LIVE_TRACKING_DEVICE_UID_KEY);
    85	
    86	  if (existing) {
    87	    return existing;
    88	  }
    89	
    90	  const generated = generateDeviceUid();
    91	
    92	  await AsyncStorage.setItem(LIVE_TRACKING_DEVICE_UID_KEY, generated);
    93	
    94	  return generated;
    95	}
