     1	import React, { useEffect, useMemo, useState } from 'react';
     2	import {
     3	  ActivityIndicator,
     4	  Alert,
     5	  Platform,
     6	  SafeAreaView,
     7	  ScrollView,
     8	  StyleSheet,
     9	  Text,
    10	  TouchableOpacity,
    11	  View,
    12	} from 'react-native';
    13	
    14	import * as Location from 'expo-location';
    15	
    16	import {
    17	  FoundationEmployee,
    18	  LocationTrackingPointFoundation,
    19	  MobileDeviceFoundation,
    20	  getFoundationEmployees,
    21	  getLatestLiveTrackingPoints,
    22	  getOrCreateMobileDeviceUid,
    23	  registerLiveTrackingDevice,
    24	  sendLiveTrackingLocation,
    25	} from '../services/liveTrackingApi';
    26	
    27	type UiStatus = 'idle' | 'loading' | 'success' | 'error';
    28	
    29	function formatEmployeeName(employee: FoundationEmployee) {
    30	  const firstName = employee.firstName || '';
    31	  const lastName = employee.lastName || '';
    32	  const fullName = `${firstName} ${lastName}`.trim();
    33	
    34	  if (fullName) {
    35	    return fullName;
    36	  }
    37	
    38	  return employee.email || employee.id;
    39	}
    40	
    41	function formatDate(value?: string) {
    42	  if (!value) {
    43	    return '-';
    44	  }
    45	
    46	  try {
    47	    return new Intl.DateTimeFormat('pt-BR', {
    48	      dateStyle: 'short',
    49	      timeStyle: 'medium',
    50	    }).format(new Date(value));
    51	  } catch {
    52	    return value;
    53	  }
    54	}
    55	
    56	export default function LiveTrackingScreen() {
    57	  const [employees, setEmployees] = useState<FoundationEmployee[]>([]);
    58	  const [selectedEmployeeId, setSelectedEmployeeId] = useState('');
    59	  const [device, setDevice] = useState<MobileDeviceFoundation | null>(null);
    60	  const [latestPoints, setLatestPoints] = useState<LocationTrackingPointFoundation[]>([]);
    61	  const [status, setStatus] = useState<UiStatus>('idle');
    62	  const [message, setMessage] = useState('Pronto para validar Live Tracking.');
    63	  const [lastLocation, setLastLocation] = useState<{
    64	    latitude: number;
    65	    longitude: number;
    66	    accuracy?: number | null;
    67	  } | null>(null);
    68	
    69	  const selectedEmployee = useMemo(
    70	    () => employees.find((employee) => employee.id === selectedEmployeeId),
    71	    [employees, selectedEmployeeId],
    72	  );
    73	
    74	  async function loadEmployees() {
    75	    setStatus('loading');
    76	    setMessage('Carregando colaboradores Foundation...');
    77	
    78	    try {
    79	      const data = await getFoundationEmployees();
    80	
    81	      setEmployees(data);
    82	
    83	      if (!selectedEmployeeId && data.length > 0) {
    84	        setSelectedEmployeeId(data[0].id);
    85	      }
    86	
    87	      setStatus('success');
    88	      setMessage('Colaboradores carregados com sucesso.');
    89	    } catch (error) {
    90	      setStatus('error');
    91	      setMessage(error instanceof Error ? error.message : 'Erro ao carregar colaboradores.');
    92	    }
    93	  }
    94	
    95	  async function refreshLatestPoints() {
    96	    setStatus('loading');
    97	    setMessage('Atualizando últimas localizações...');
    98	
    99	    try {
   100	      const data = await getLatestLiveTrackingPoints();
   101	
   102	      setLatestPoints(data);
   103	      setStatus('success');
   104	      setMessage('Últimas localizações atualizadas.');
   105	    } catch (error) {
   106	      setStatus('error');
   107	      setMessage(error instanceof Error ? error.message : 'Erro ao atualizar localizações.');
   108	    }
   109	  }
   110	
   111	  async function handleRegisterDevice() {
   112	    if (!selectedEmployeeId) {
   113	      setStatus('error');
   114	      setMessage('Selecione um colaborador antes de registrar o dispositivo.');
   115	      return null;
   116	    }
   117	
   118	    setStatus('loading');
   119	    setMessage('Registrando dispositivo mobile...');
   120	
   121	    try {
   122	      const deviceUid = await getOrCreateMobileDeviceUid();
   123	
   124	      const registeredDevice = await registerLiveTrackingDevice({
   125	        employeeProfileId: selectedEmployeeId,
   126	        deviceUid,
   127	        platform: Platform.OS,
   128	        model: String(Platform.Version || 'mobile'),
   129	        appVersion: 'mobile-validation',
   130	      });
   131	
   132	      setDevice(registeredDevice);
   133	      setStatus('success');
   134	      setMessage('Dispositivo mobile registrado com sucesso.');
   135	
   136	      return registeredDevice;
   137	    } catch (error) {
   138	      setStatus('error');
   139	      setMessage(error instanceof Error ? error.message : 'Erro ao registrar dispositivo.');
   140	      return null;
   141	    }
   142	  }
   143	
   144	  async function requestLocationPermission() {
   145	    const permission = await Location.requestForegroundPermissionsAsync();
   146	
   147	    if (permission.status !== 'granted') {
   148	      throw new Error('Permissão de localização não concedida.');
   149	    }
   150	  }
   151	
   152	  async function handleSendCurrentLocation() {
   153	    if (!selectedEmployeeId) {
   154	      setStatus('error');
   155	      setMessage('Selecione um colaborador antes de enviar localização.');
   156	      return;
   157	    }
   158	
   159	    setStatus('loading');
   160	    setMessage('Solicitando permissão de localização...');
   161	
   162	    try {
   163	      await requestLocationPermission();
   164	
   165	      const activeDevice = device || (await handleRegisterDevice());
   166	
   167	      if (!activeDevice) {
   168	        return;
   169	      }
   170	
   171	      setMessage('Capturando localização atual...');
   172	
   173	      const current = await Location.getCurrentPositionAsync({
   174	        accuracy: Location.Accuracy.High,
   175	      });
   176	
   177	      const latitude = current.coords.latitude;
   178	      const longitude = current.coords.longitude;
   179	      const accuracy = current.coords.accuracy;
   180	
   181	      setLastLocation({
   182	        latitude,
   183	        longitude,
   184	        accuracy,
   185	      });
   186	
   187	      setMessage('Enviando ponto GPS para a Foundation...');
   188	
   189	      await sendLiveTrackingLocation({
   190	        employeeProfileId: selectedEmployeeId,
   191	        deviceId: activeDevice.id,
   192	        latitude,
   193	        longitude,
   194	        accuracyMeters: accuracy || undefined,
   195	        source: 'MOBILE',
   196	        trackedAt: new Date().toISOString(),
   197	        metadata: {
   198	          origem: 'mobile-app',
   199	          tela: 'LiveTrackingScreen',
   200	          platform: Platform.OS,
   201	        },
   202	      });
   203	
   204	      setStatus('success');
   205	      setMessage('Localização enviada com sucesso.');
   206	
   207	      await refreshLatestPoints();
   208	    } catch (error) {
   209	      setStatus('error');
   210	      setMessage(error instanceof Error ? error.message : 'Erro ao enviar localização.');
   211	      Alert.alert(
   212	        'Live Tracking',
   213	        error instanceof Error ? error.message : 'Erro ao enviar localização.',
   214	      );
   215	    }
   216	  }
   217	
   218	  useEffect(() => {
   219	    loadEmployees();
   220	    refreshLatestPoints();
   221	    // eslint-disable-next-line react-hooks/exhaustive-deps
   222	  }, []);
   223	
   224	  const statusStyle =
   225	    status === 'error'
   226	      ? styles.statusError
   227	      : status === 'success'
   228	        ? styles.statusSuccess
   229	        : status === 'loading'
   230	          ? styles.statusLoading
   231	          : styles.statusIdle;
   232	
   233	  return (
   234	    <SafeAreaView style={styles.safe}>
   235	      <ScrollView contentContainerStyle={styles.container}>
   236	        <View style={styles.header}>
   237	          <Text style={styles.kicker}>Foundation</Text>
   238	          <Text style={styles.title}>Live Tracking</Text>
   239	          <Text style={styles.subtitle}>
   240	            Tela manual de validação mobile sem rastreamento em segundo plano.
   241	          </Text>
   242	        </View>
   243	
   244	        <View style={[styles.statusBox, statusStyle]}>
   245	          {status === 'loading' ? <ActivityIndicator /> : null}
   246	          <Text style={styles.statusText}>{message}</Text>
   247	        </View>
   248	
   249	        <View style={styles.card}>
   250	          <Text style={styles.cardTitle}>1. Selecionar colaborador</Text>
   251	
   252	          <TouchableOpacity style={styles.secondaryButton} onPress={loadEmployees}>
   253	            <Text style={styles.secondaryButtonText}>Recarregar colaboradores</Text>
   254	          </TouchableOpacity>
   255	
   256	          {employees.length === 0 ? (
   257	            <Text style={styles.muted}>Nenhum colaborador carregado.</Text>
   258	          ) : null}
   259	
   260	          {employees.map((employee) => {
   261	            const selected = employee.id === selectedEmployeeId;
   262	
   263	            return (
   264	              <TouchableOpacity
   265	                key={employee.id}
   266	                style={[styles.employeeItem, selected ? styles.employeeItemSelected : null]}
   267	                onPress={() => {
   268	                  setSelectedEmployeeId(employee.id);
   269	                  setDevice(null);
   270	                }}
   271	              >
   272	                <Text style={styles.employeeName}>{formatEmployeeName(employee)}</Text>
   273	                <Text style={styles.employeeMeta}>{employee.email || employee.id}</Text>
   274	                <Text style={styles.employeeMeta}>Status: {employee.status || '-'}</Text>
   275	              </TouchableOpacity>
   276	            );
   277	          })}
   278	        </View>
   279	
   280	        <View style={styles.card}>
   281	          <Text style={styles.cardTitle}>2. Dispositivo</Text>
   282	
   283	          {selectedEmployee ? (
   284	            <Text style={styles.muted}>
   285	              Colaborador selecionado: {formatEmployeeName(selectedEmployee)}
   286	            </Text>
   287	          ) : (
   288	            <Text style={styles.muted}>Selecione um colaborador.</Text>
   289	          )}
   290	
   291	          <TouchableOpacity style={styles.primaryButton} onPress={handleRegisterDevice}>
   292	            <Text style={styles.primaryButtonText}>Registrar dispositivo mobile</Text>
   293	          </TouchableOpacity>
   294	
   295	          {device ? (
   296	            <View style={styles.infoBox}>
   297	              <Text style={styles.infoText}>Device ID: {device.id}</Text>
   298	              <Text style={styles.infoText}>Device UID: {device.deviceUid}</Text>
   299	            </View>
   300	          ) : null}
   301	        </View>
   302	
   303	        <View style={styles.card}>
   304	          <Text style={styles.cardTitle}>3. Localização atual</Text>
   305	
   306	          <TouchableOpacity style={styles.dangerButton} onPress={handleSendCurrentLocation}>
   307	            <Text style={styles.primaryButtonText}>Capturar e enviar localização</Text>
   308	          </TouchableOpacity>
   309	
   310	          {lastLocation ? (
   311	            <View style={styles.infoBox}>
   312	              <Text style={styles.infoText}>Latitude: {lastLocation.latitude}</Text>
   313	              <Text style={styles.infoText}>Longitude: {lastLocation.longitude}</Text>
   314	              <Text style={styles.infoText}>Precisão: {lastLocation.accuracy ?? '-'} m</Text>
   315	            </View>
   316	          ) : null}
   317	        </View>
   318	
   319	        <View style={styles.card}>
   320	          <Text style={styles.cardTitle}>4. Últimas localizações</Text>
   321	
   322	          <TouchableOpacity style={styles.secondaryButton} onPress={refreshLatestPoints}>
   323	            <Text style={styles.secondaryButtonText}>Atualizar últimas localizações</Text>
   324	          </TouchableOpacity>
   325	
   326	          {latestPoints.length === 0 ? (
   327	            <Text style={styles.muted}>Nenhuma localização encontrada.</Text>
   328	          ) : null}
   329	
   330	          {latestPoints.map((point) => (
   331	            <View key={point.id} style={styles.pointItem}>
   332	              <Text style={styles.pointTitle}>{point.employeeProfileId || '-'}</Text>
   333	              <Text style={styles.pointMeta}>
   334	                {point.latitude}, {point.longitude}
   335	              </Text>
   336	              <Text style={styles.pointMeta}>Origem: {point.source || '-'}</Text>
   337	              <Text style={styles.pointMeta}>Data: {formatDate(point.trackedAt)}</Text>
   338	            </View>
   339	          ))}
   340	        </View>
   341	      </ScrollView>
   342	    </SafeAreaView>
   343	  );
   344	}
