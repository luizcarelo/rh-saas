     1	import React, { useState, useEffect } from 'react';
     2	import { View, Text, StyleSheet, TouchableOpacity, ActivityIndicator, Alert, Image } from 'react-native';
     3	import * as Location from 'expo-location';
     4	import AsyncStorage from '@react-native-async-storage/async-storage';
     5	import api from '../services/api';
     6	import { LogOut, Clock, MapPin } from 'lucide-react-native';
     7	
     8	export default function HomeScreen({ navigation }: any) {
     9	  const [loading, setLoading] = useState(false);
    10	  const [currentTime, setCurrentTime] = useState(new Date());
    11	
    12	  useEffect(() => {
    13	    const timer = setInterval(() => setCurrentTime(new Date()), 1000);
    14	    return () => clearInterval(timer);
    15	  }, []);
    16	
    17	  const handleLogout = async () => {
    18	    await AsyncStorage.removeItem('@LH_Token');
    19	    navigation.replace('Login');
    20	  };
    21	
    22	  const handleClockIn = async () => {
    23	    setLoading(true);
    24	    try {
    25	      let { status } = await Location.requestForegroundPermissionsAsync();
    26	      if (status !== 'granted') {
    27	        Alert.alert('Erro', 'Acesso ao GPS é obrigatório.');
    28	        setLoading(false);
    29	        return;
    30	      }
    31	
    32	      const location = await Location.getCurrentPositionAsync({ accuracy: Location.Accuracy.High });
    33	      const { latitude, longitude } = location.coords;
    34	
    35	      await api.post('/v1/time-records/clock-in', { latitude, longitude });
    36	      
    37	      Alert.alert('Sucesso', 'Ponto registrado com sucesso!');
    38	    } catch (error) {
    39	      Alert.alert('Erro', 'Não foi possível salvar o ponto.');
    40	    } finally {
    41	      setLoading(false);
    42	    }
    43	  };
    44	
    45	  return (
    46	    <View style={styles.container}>

    51	
    52	      <View style={styles.clockContainer}>
    53	        <Clock color="#059669" size={40} />
    54	        <Text style={styles.timeText}>{currentTime.toLocaleTimeString('pt-BR')}</Text>

    56	      </View>
    57	
    58	      <TouchableOpacity style={[styles.mainButton, loading && styles.disabled]} onPress={handleClockIn} disabled={loading}>

    31	function formatApiError(error: unknown) {
    32	  if (error instanceof AxiosError) {
    33	    const status = error.response?.status;
    34	    const data = error.response?.data;
    35	
    36	    if (data && typeof data === 'object') {
    37	      const apiData = data as {
    38	        message?: string | string[];
    39	        error?: string;
    40	        statusCode?: number;
    41	      };
    42	
    43	      const message = Array.isArray(apiData.message)
    44	        ? apiData.message.join(', ')
    45	        : apiData.message;
    46	
    47	      return [
    48	        status ? `HTTP ${status}` : 'Erro HTTP',
    49	        message || apiData.error || error.message,
    50	      ].filter(Boolean).join(' - ');
    51	    }
    52	
    53	    return status ? `HTTP ${status} - ${error.message}` : error.message;
    54	  }
    55	
    56	  return error instanceof Error ? error.message : 'Erro inesperado.';
    57	}

    66	  }

    68	  return employee.email || employee.id;

    71	function formatDate(value?: string) {
    72	  if (!value) {
    73	    return '-';
    74	  }

    76	  try {
    77	    return new Intl.DateTimeFormat('pt-BR', {
    78	      dateStyle: 'short',
    79	      timeStyle: 'medium',
    80	    }).format(new Date(value));
    81	  } catch {
    82	    return value;
    83	  }
