    78	function normalizeErrorMessage(data: unknown, fallback: string) {
    79	  if (!data || typeof data !== "object") {
    80	    return fallback;
    81	  }
    82	
    83	  const apiError = data as ApiError;
    84	
    85	  if (Array.isArray(apiError.message)) {
    86	    return apiError.message.join(", ");
    87	  }
    88	
    89	  return apiError.message || apiError.error || fallback;
    90	}

    92	export async function apiRequest<T>(
    93	  endpoint: string,
    94	  options: RequestInit = {},
    95	): Promise<T> {
    96	  const token = getToken();
    97	
    98	  const headers: Record<string, string> = {
    99	    "Content-Type": "application/json",
   100	    ...((options.headers as Record<string, string>) || {}),
   101	  };
   102	
   103	  if (token) {
   104	    headers.Authorization = `Bearer ${token}`;
   105	  }
   106	
   107	  const response = await fetch(`${API_BASE}${endpoint}`, {
   108	    ...options,
   109	    headers,
   110	  });
   111	
   112	  const text = await response.text();
   113	
   114	  let data: unknown = null;
   115	
   116	  if (text) {
   117	    try {
   118	      data = JSON.parse(text);
   119	    } catch {
   120	      data = text;
   121	    }
   122	  }
   123	
   124	  if (response.status === 401) {
   125	    clearSession();
   126	  }
   127	
   128	  if (!response.ok) {
   129	    throw new Error(
   130	      normalizeErrorMessage(data, `Erro HTTP ${response.status}`),
   131	    );
   132	  }
   133	
   134	  return data as T;
   135	}

   157	async function requestJson(endpoint: string) {
   158	  const token = getSuperAdminAuthToken();
   159	
   160	  if (!token) {
   161	    throw new Error('Token de autenticação não encontrado.');
   162	  }
   163	
   164	  const baseUrl = getApiBaseUrl();
   165	  const url = `${baseUrl}${endpoint}`;
   166	
   167	  const response = await fetch(url, {
   168	    method: 'GET',
   169	    headers: {
   170	      Authorization: `Bearer ${token}`,
   171	      'Content-Type': 'application/json',
   172	    },
   173	  });
   174	
   175	  if (response.status === 401) {
   176	    clearSessionAndRedirect();
   177	    throw new Error('Sessão expirada.');
   178	  }
   179	
   180	  const text = await response.text();
   181	
   182	  let payload: unknown = null;
   183	
   184	  if (text) {
   185	    try {
   186	      payload = JSON.parse(text) as unknown;
   187	    } catch {
   188	      payload = text;
   189	    }
   190	  }
   191	
   192	  if (!response.ok) {
   193	    const message =
   194	      typeof payload === 'object' && payload !== null
   195	        ? String((payload as UnknownRecord).message || response.statusText)
   196	        : response.statusText;
   197	
   198	    throw new Error(message || 'Falha ao buscar clientes SaaS.');
   199	  }
   200	
   201	  return payload as ListResponseShape;
   202	}
