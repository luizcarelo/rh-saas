     8	export class ClockEventsService {
     9	  constructor(
    10	    @InjectRepository(ClockEvent)
    11	    private readonly clockEventRepository: Repository<ClockEvent>,
    12	  ) {}
    13	
    14	  /**
    15	   * 1. GEOFENCING: Calcula a distância entre a batida e o local autorizado
    16	   * Retorna a distância em metros (Fórmula de Haversine)
    17	   */
    18	  private calculateDistance(lat1: number, lon1: number, lat2: number, lon2: number): number {
    19	    const R = 6371e3; // Raio da Terra em metros
    20	    const phi1 = (lat1 * Math.PI) / 180;
    21	    const phi2 = (lat2 * Math.PI) / 180;
    22	    const deltaPhi = ((lat2 - lat1) * Math.PI) / 180;
    23	    const deltaLambda = ((lon2 - lon1) * Math.PI) / 180;
    24	
    25	    const a =
    26	      Math.sin(deltaPhi / 2) * Math.sin(deltaPhi / 2) +
    27	      Math.cos(phi1) * Math.cos(phi2) * Math.sin(deltaLambda / 2) * Math.sin(deltaLambda / 2);
    28	    
    29	    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    30	    return R * c; 
    31	  }
    32	
    33	  /**
    34	   * 2. ASSINATURA MTE: Gera o Hash SHA-256 para garantir imutabilidade
    35	   */
    36	  private generateSecurityHash(data: {
    37	    tenantId: string;
    38	    employeeId: string;
    39	    timestamp: string;
    40	    eventType: string;
    41	    latitude: number;
    42	    longitude: number;
    43	  }): string {
    44	    // A ordem dos fatores importa para o Hash. Padrão estrito de concatenação.
    45	    const rawString = `${data.tenantId}|${data.employeeId}|${data.timestamp}|${data.eventType}|${data.latitude}|${data.longitude}`;
    46	    return crypto.createHash('sha256').update(rawString).digest('hex');
    47	  }
    48	
    49	  /**
    50	   * 3. REGISTRO OFICIAL DO PONTO (Chamado pela API / App do Colaborador)
    51	   */
    52	  async registerClock(data: {
    53	    tenantId: string;
    54	    employeeId: string;
    55	    eventType: ClockEventType;
    56	    latitude: number;
    57	    longitude: number;
    58	    allowedLatitude: number;
    59	    allowedLongitude: number;
    60	    allowedRadius: number; // Ex: 100 metros
    61	    deviceInfo?: string;
    62	    clientIp?: string;
    63	  }): Promise<ClockEvent> {
    64	    
    65	    // Calcula se o funcionário está dentro da área permitida
    66	    const distance = this.calculateDistance(
    67	      data.latitude,
    68	      data.longitude,
    69	      data.allowedLatitude,
    70	      data.allowedLongitude
    71	    );
    72	    
    73	    const isInsideGeofence = distance <= data.allowedRadius;
    74	
    75	    // Regra de Negócio: Se a empresa não permite bater ponto fora da cerca, barra aqui.
    76	    // (Em um SaaS, isso geralmente é uma flag configurável no Painel do RH).
    77	    // if (!isInsideGeofence && !tenantConfig.allowOutlyingClockIn) {
    78	    //   throw new BadRequestException('Sua localização atual está fora do raio permitido pela empresa.');
    79	    // }
    80	
    81	    // Congela o tempo exato (UTC) da transação para o MTE
    82	    const currentTimestamp = new Date().toISOString(); 
    83	    
    84	    // Gera a assinatura inviolável
    85	    const rowHash = this.generateSecurityHash({
    86	      tenantId: data.tenantId,
    87	      employeeId: data.employeeId,
    88	      timestamp: currentTimestamp,
    89	      eventType: data.eventType,

    91	      longitude: data.longitude
    92	    });

    97	      employeeId: data.employeeId,

    99	      latitude: data.latitude,
   100	      longitude: data.longitude,
   101	      isInsideGeofence,
   102	      deviceInfo: data.deviceInfo,
   103	      clientIp: data.clientIp,

   282	  async getExpectedMinutesForDate(
   283	    tenantId: string,
   284	    employeeId: string,
   285	    targetDateStr: string,
   286	  ): Promise<{
   287	    expected: number;
   288	    tolerance: number;
   289	  }> {
   290	
   291	    const targetDate =
   292	      new Date(

   294	      );

   296	    const link =
   297	      await this.employeeScheduleRepo
   298	        .createQueryBuilder('es')
   299	        .leftJoinAndSelect(

   310	        )
   311	        .andWhere(
   312	          'es.startDate <= :targetDate',
   313	          { targetDate: targetDateStr },
   314	        )
   315	        .andWhere(
   316	          '(es.endDate IS NULL OR es.endDate >= :targetDate)',
   317	          { targetDate: targetDateStr },
   318	        )
   319	        .getOne();
   320	
   321	    if (!link) {
   322	      return {
   323	        expected: 0,
   324	        tolerance: 0,
   325	      };
   326	    }
   327	
   328	    const { schedule } = link;
   329	
   330	    if (
   331	      schedule.type ===
   332	      ScheduleType.FIXED
   333	    ) {
   334	      const dayOfWeek =
   335	        targetDate.getUTCDay();
   336	
   337	      if (
   338	        schedule.workDays &&
   339	        schedule.workDays.includes(
   340	          dayOfWeek,
   341	        )
   342	      ) {
   343	        return {
   344	          expected:
   345	            schedule.expectedMinutesPerDay,
   346	          tolerance:
   347	            schedule.dailyToleranceMinutes,
   348	        };
   349	      }
   350	
   351	      return {
   352	        expected: 0,
   353	        tolerance:
   354	          schedule.dailyToleranceMinutes,
   355	      };
   356	    }
   357	
   358	    if (
   359	      schedule.type ===
   360	      ScheduleType.SHIFT_12X36
   361	    ) {
   362	      const startDate =
   363	        new Date(
   364	          `${link.startDate}T12:00:00Z`,
   365	        );
   366	
   367	      const diffTime =
   368	        Math.abs(
   369	          targetDate.getTime() -
   370	          startDate.getTime(),
   371	        );
   372	
   373	      const diffDays =
   374	        Math.floor(

    79	  /**
    80	   * Processa o saldo do dia cruzando os pontos batidos com a jornada prevista
    81	   */
    82	  async processDailyBalance(tenantId: string, employeeId: string, targetDate: string, expectedMinutes: number) {
    83	    // 1. Define o início e fim do dia para buscar as batidas no banco
    84	    const startOfDay = new Date(`${targetDate}T00:00:00.000Z`);
    85	    const endOfDay = new Date(`${targetDate}T23:59:59.999Z`);
    86	
    87	    // 2. Busca todas as batidas imutáveis do funcionário neste dia
    88	    const events = await this.clockEventRepo.find({
    89	      where: {
    90	        tenantId,
    91	        employeeId,
    92	        timestamp: Between(startOfDay, endOfDay)
    93	      },
    94	      order: { timestamp: 'ASC' } // Ordena cronologicamente
    95	    });
    96	
    97	    let workedMinutes = 0;
    98	    let lastEntryTime: Date | null = null;
    99	
   100	    // 3. Lógica pareada de cálculo (Entrada -> Saída)
   101	    for (const event of events) {
   102	      if (event.eventType === ClockEventType.ENTRADA || event.eventType === ClockEventType.RETORNO_INTERVALO) {
   103	        lastEntryTime = event.timestamp;
   104	      } 
   105	      else if (event.eventType === ClockEventType.SAIDA_INTERVALO || event.eventType === ClockEventType.SAIDA) {
   106	        if (lastEntryTime) {
   107	          workedMinutes += this.getMinutesBetween(lastEntryTime, event.timestamp);
   108	          lastEntryTime = null; // Reseta para o próximo par
   109	        }
   110	      }
   111	    }
   112	
   113	    // 4. Calcula o saldo final (Matemática Trabalhista)
   114	    const balanceMinutes = workedMinutes - expectedMinutes;
   115	
   116	    // 5. Salva ou atualiza o banco de horas do dia
   117	    let timeBankRecord = await this.timeBankRepo.findOne({
   118	      where: { tenantId, employeeId, date: targetDate }
   119	    });
   120	
   121	    if (!timeBankRecord) {
   122	      timeBankRecord = this.timeBankRepo.create({
   123	        tenantId,
   124	        employeeId,
   125	        date: targetDate,
   126	        expectedMinutes,
   127	      });
   128	    }
   129	
   130	    timeBankRecord.workedMinutes = workedMinutes;
   131	    timeBankRecord.balanceMinutes = balanceMinutes;

     1	import { Injectable } from '@nestjs/common';
     2	import { TimeRecord } from './time-record.entity';
     3	
     4	@Injectable()
     5	export class TimeCalculatorService {
     6	  
     7	  // Converte "HH:mm" para minutos totais (Ex: "08:30" -> 510)
     8	  timeToMinutes(timeStr: string): number {
     9	    if (!timeStr) return 0;
    10	    const [hours, minutes] = timeStr.split(':').map(Number);
    11	    return (hours * 60) + minutes;
    12	  }
    13	
    14	  // Converte minutos para string formatada "+HH:mm" ou "-HH:mm"
    15	  minutesToBalanceString(totalMinutes: number): string {
    16	    const sign = totalMinutes >= 0 ? '+' : '-';
    17	    const absMinutes = Math.abs(totalMinutes);
    18	    const hours = Math.floor(absMinutes / 60).toString().padStart(2, '0');
    19	    const mins = (absMinutes % 60).toString().padStart(2, '0');
    20	    return `${sign}${hours}:${mins}`;
    21	  }
    22	
    23	  // Recebe as batidas do dia e calcula o saldo baseado na escala
    24	  calculateDailyBalance(records: TimeRecord[], expectedMinutes: number) {
    25	    if (records.length === 0) {
    26	      return { status: 'falta', balanceMinutes: -expectedMinutes, balanceStr: this.minutesToBalanceString(-expectedMinutes) };
    27	    }
    28	
    29	    // Ordena as batidas cronologicamente
    30	    const sorted = records.sort((a, b) => new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime());
    31	    
    32	    let workedMinutes = 0;
    33	
    34	    // Lógica para pares de batidas (Entrada 1 -> Saída 1, Entrada 2 -> Saída 2)
    35	    for (let i = 0; i < sorted.length; i += 2) {
    36	      if (sorted[i + 1]) {
    37	        const inTime = new Date(sorted[i].timestamp).getTime();
    38	        const outTime = new Date(sorted[i + 1].timestamp).getTime();
    39	        const diffMs = outTime - inTime;
    40	        workedMinutes += Math.floor(diffMs / 60000); // Converte MS para minutos
    41	      }
    42	    }
    43	
    44	    // Identifica se esqueceu de bater a saída (ímpar)
    45	    if (sorted.length % 2 !== 0) {
    46	      return { status: 'incompleto', balanceMinutes: 0, balanceStr: 'Erro: Batida ímpar', workedMinutes };
    47	    }
    48	
    49	    const balanceMinutes = workedMinutes - expectedMinutes;
    50	    const balanceStr = this.minutesToBalanceString(balanceMinutes);
    51	    
    52	    // Define o status (com tolerância de 10 minutos por dia segundo a CLT)
    53	    let status = 'ok';
    54	    if (balanceMinutes < -10) status = 'atraso';
    55	    else if (balanceMinutes > 10) status = 'extra';
    56	
    57	    return { status, balanceMinutes, balanceStr, workedMinutes };
    58	  }
    59	}

### /opt/rh-saas/backend/src/modules/users/users.service.ts
     1	import { AuditService } from '../audit/audit.service';
     2	import { Injectable, ConflictException } from '@nestjs/common';
     3	import { InjectRepository } from '@nestjs/typeorm';
     4	import { Repository } from 'typeorm';
     5	import * as bcrypt from 'bcrypt';
     6	
     7	import { User } from './user.entity';
     8	import { CreateUserDto } from './dto/create-user.dto';
     9	
    10	@Injectable()
    11	export class UsersService {
    12	
    13	  constructor(
    14	    @InjectRepository(User)
    15	    private readonly userRepo: Repository<User>,
    16	
    17	    private readonly auditService: AuditService,
    18	  ) {}
    19	
    20	  async create(data: CreateUserDto) {
    21	
    22	    const existing = await this.userRepo.findOne({
    23	      where: {
    24	        email: data.email,
    25	      },
    26	    });
    27	
    28	    if (existing) {
    29	      throw new ConflictException(
    30	        'Usuário já cadastrado.',
    31	      );
    32	    }
    33	
    34	    const passwordHash = await bcrypt.hash(
    35	      data.password,
    36	      10,
    37	    );
    38	
    39	    const user = this.userRepo.create({
    40	      tenantId: data.tenantId,
    41	      email: data.email,
    42	      passwordHash,
    43	      isActive: true,
    44	    });
    45	
    46	    const savedUser =
    47	      await this.userRepo.save(user);
    48	
    49	    await this.auditService.create({
    50	      tenantId: savedUser.tenantId,
    51	      entityId: savedUser.id,
    52	      action: 'USER_CREATED',
    53	      entityType: 'User',
    54	      details: savedUser.email,
    55	    });
    56	
    57	    return savedUser;
    58	  }
    59	
    60	  async findByEmail(email: string) {
    61	    return this.userRepo.findOne({
    62	      where: { email },
    63	    });
    64	  }
    65	
    66	  async findById(id: string) {
    67	    return this.userRepo.findOne({
    68	      where: { id },
    69	    });
    70	  }
    71	
    72	  async findAll() {
    73	
    74	    const users = await this.userRepo.find({
    75	      order: {
    76	        email: 'ASC',
    77	      },
    78	    });
    79	
    80	    return users.map(
    81	      ({ passwordHash, ...user }) => user,
    82	    );
    83	  }
    84	}
